    def initialize(subdomain, options = {})
      options = { :ssl => false }.merge(options)
      @subdomain = subdomain
      @uri = URI.parse("#{options[:ssl] ? "https" : "http"}://#{subdomain}.campfirenow.com")
      @room = nil
      @directory = "."
      @agent = WWW::Mechanize.new
      @logged_in = false
    end
